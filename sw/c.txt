#define AES_BASE  AES_CORE_0_BASE
#define WORD_TO_BYTE(w) ((w) * 4)

enum {
    KEY_WORD0   = 0,
    PLAIN_WORD0 = 4,
    START_WORD  = 8
};

static void write_u32s(uint32_t base, int word_idx, const uint32_t *v) {
    int i;
    for (i = 0; i < 4; ++i) {
        IOWR_32DIRECT(base, WORD_TO_BYTE(word_idx + i), v[i]);
    }
}

static void read_u32s(uint32_t base, int word_idx, uint32_t *v) {
    int i;
    for (i = 0; i < 4; ++i) {
        v[i] = IORD_32DIRECT(base, WORD_TO_BYTE(word_idx + i));
    }
}

static inline void write_start(uint32_t base, uint8_t v) {
    IOWR_32DIRECT(base, WORD_TO_BYTE(START_WORD), (uint32_t)(v & 1u));
}

void aes_encrypt(const uint32_t key[4], const uint32_t pt[4], uint32_t ct[4]) {
    write_u32s(AES_BASE, KEY_WORD0, key);
    write_u32s(AES_BASE, PLAIN_WORD0, pt);
    write_start(AES_BASE, 1);
    write_start(AES_BASE, 0);
    read_u32s(AES_BASE, KEY_WORD0, ct);
}
